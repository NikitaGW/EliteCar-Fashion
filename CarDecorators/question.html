Strings // Reverse a string function reverseString(str) { return str.split('').reverse().join(''); } // Check if a string is a palindrome function isPalindrome(str) { const reversed = str.split('').reverse().join(''); return str === reversed; } // Count vowels in a string function countVowels(str) { return (str.match(/[aeiou]/gi) || []).length; } // Capitalize the first letter of each word function capitalizeWords(str) { return str.replace(/\b\w/g, c => c.toUpperCase()); } // Truncate a string to a given length function truncate(str, length) { return str.length > length ? str.slice(0, length) + '...' : str; } // Convert a string to camelCase function toCamelCase(str) { return str .toLowerCase() .replace(/[^a-zA-Z0-9]+(.)/g, (_, chr) => chr.toUpperCase()); } // Check if two strings are anagrams function isAnagram(str1, str2) { const sort = str => str.split('').sort().join(''); return sort(str1) === sort(str2); } // Remove duplicate characters from a string function removeDuplicates(str) { return [...new Set(str)].join(''); } // Count occurrences of a character in a string function countChar(str, char) { return str.split(char).length - 1; } // Reverse words in a sentence function reverseWords(sentence) { return sentence.split(' ').reverse().join(' '); } // ✅ 2. Numbers // Check if a number is prime function isPrime(num) { if (num <= 1) return false; for (let i = 2; i <= Math.sqrt(num); i++) { if (num % i === 0) return false; } return true; } // Return the factorial of a number function factorial(n) { return n === 0 ? 1 : n * factorial(n - 1); } // Generate Fibonacci sequence up to n terms function fibonacci(n) { let seq = [0, 1]; for (let i = 2; i < n; i++) { seq.push(seq[i - 1] + seq[i - 2]); } return seq.slice(0, n); } // Check if a number is even or odd function isEven(num) { return num % 2 === 0 ? 'Even' : 'Odd'; } // Convert Celsius to Fahrenheit function celsiusToFahrenheit(c) { return (c * 9) / 5 + 32; } // Find the largest among three numbers function largest(a, b, c) { return Math.max(a, b, c); } // Sum digits of a number function sumDigits(num) { return num.toString().split('').reduce((a, b) => a + +b, 0); } // Check if a number is an Armstrong number function isArmstrong(num) { let sum = 0; let temp = num; let digits = num.toString().length; while (temp > 0) { let d = temp % 10; sum += d ** digits; temp = Math.floor(temp / 10); } return sum === num; } // Check if a year is a leap year function isLeapYear(year) { return (year % 4 === 0 && year % 100 !== 0) || year % 400 === 0; } // Find the power of a number (without Math.pow()) function power(base, exponent) { let result = 1; for (let i = 0; i < exponent; i++) { result *= base; } return result; } // ✅ 3. Arrays // Find the largest number in an array function largestInArray(arr) { return Math.max(...arr); } // Remove duplicates from an array function removeArrayDuplicates(arr) { return [...new Set(arr)]; } // Sort an array of numbers (ascending) function sortArray(arr) { return arr.sort((a, b) => a - b); } // Count even and odd numbers in an array function countEvenOdd(arr) { let even = 0, odd = 0; arr.forEach(n => n % 2 === 0 ? even++ : odd++); return { even, odd }; } // Sum all numbers in an array function sumArray(arr) { return arr.reduce((a, b) => a + b, 0); } // Find the index of a given element in an array function findIndex(arr, val) { return arr.indexOf(val); } // Reverse an array function reverseArray(arr) { return arr.reverse(); } // Find the second largest number in an array function secondLargest(arr) { let sorted = [...new Set(arr)].sort((a, b) => b - a); return sorted[1]; } // ✅ 4. Objects // Count keys in an object function countKeys(obj) { return Object.keys(obj).length; } // Merge two objects function mergeObjects(obj1, obj2) { return { ...obj1, ...obj2 }; } // Find the key with the highest value function keyWithMaxValue(obj) { return Object.keys(obj).reduce((a, b) => obj[a] > obj[b] ? a : b); } // Convert an object to an array of key-value pairs function objectToArray(obj) { return Object.entries(obj); } // Clone an object function cloneObject(obj) { return { ...obj }; } // Check if a key exists in an object function keyExists(obj, key) { return key in obj; } // Sum all numeric values in an object function sumObjectValues(obj) { return Object.values(obj).reduce((a, b) => typeof b === 'number' ? a + b : a, 0); } // Swap keys and values in an object function swapKeysValues(obj) { const swapped = {}; for (let key in obj) { swapped[obj[key]] = key; } return swapped; } // Create an object from two arrays (keys + values) function arraysToObject(keys, values) { const obj = {}; keys.forEach((key, i) => obj[key] = values[i]); return obj; } // Remove properties with null or undefined values function cleanObject(obj) { Object.keys(obj).forEach(key => { if (obj[key] == null) delete obj[key]; }); return obj; } // ✅ 5.
